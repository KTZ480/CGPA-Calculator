<div id="result"></div>

<script>
  function generateInputs() {
    const count = parseInt(document.getElementById('subjectCount').value);
    const form = document.getElementById('cgpaForm');
    form.innerHTML = '';

    for (let i = 0; i < count; i++) {
      const row = document.createElement('div');
      row.className = 'subject-row';

      const credit = document.createElement('input');
      credit.type = 'number';
      credit.placeholder = 'ক্রেডিট';
      credit.required = true;

      const grade = document.createElement('input');
      grade.type = 'number';
      grade.placeholder = 'গ্রেড (উদাহরণ: 4, 3.5)';
      grade.step = '0.01';
      grade.required = true;

      row.appendChild(credit);
      row.appendChild(grade);
      form.appendChild(row);
    }

    const calcBtn = document.createElement('button');
    calcBtn.type = 'button';
    calcBtn.textContent = 'CGPA হিসাব করুন';
    calcBtn.onclick = calculateCGPA;

    form.appendChild(calcBtn);
  }

  function calculateCGPA() {
    const rows = document.querySelectorAll('.subject-row');
    let totalCredits = 0;
    let totalPoints = 0;

    for (const row of rows) {
      const credit = parseFloat(row.children[0].value);
      const grade = parseFloat(row.children[1].value);

      if (isNaN(credit) || isNaN(grade)) continue;

      totalCredits += credit;
      totalPoints += credit * grade;
    }

    const cgpa = totalPoints / totalCredits;
    document.getElementById('result').innerText = 
      `আপনার CGPA: ${cgpa.toFixed(2)}`;
  }
</script>
